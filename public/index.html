const readline = require("readline");
const { execSync } = require("child_process");
const fs = require("fs");
const https = require("https");
const path = require("path");

// === CONFIG ===
const API_SERVER = "https://iceymd.onrender.com/api/auth-folder"; 
const GITHUB_REPO = "https://github.com/asoa1/Icey-md-bot.git"; // replace
const BOT_FOLDER = "icey-md-bot";

// === readline ===
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout,
});

// === ensure adm-zip installed ===
function ensureAdmZip() {
  try {
    require.resolve("adm-zip");
  } catch (e) {
    console.log("📦 Installing adm-zip...");
    execSync("npm install adm-zip", { stdio: "inherit" });
  }
}
ensureAdmZip();
const AdmZip = require("adm-zip");

// === download to buffer helper ===
function downloadBuffer(url) {
  return new Promise((resolve, reject) => {
    https.get(url, (res) => {
      if (res.statusCode !== 200) {
        reject(new Error(`Failed to get '${url}' (${res.statusCode})`));
        return;
      }
      const data = [];
      res.on("data", (chunk) => data.push(chunk));
      res.on("end", () => resolve(Buffer.concat(data)));
    }).on("error", reject);
  });
}

// === MAIN ===
rl.question(
  "❖ ICEY SETUP ❖\n╭────────────────╼\n╎ Enter Session Name : \n╰────────────────╼ ",
  async (sessionName) => {
    try {
      console.log(`\n🔄 Setting up with session: ${sessionName}...\n`);

      // Step 1: Pull session zip from API (as buffer)
      const sessionUrl = `${API_SERVER}/${sessionName}`;
      console.log("⬇️ Downloading auth_folder from API...");
      const buffer = await downloadBuffer(sessionUrl);

      // Step 2: Extract directly from buffer
      console.log("📂 Extracting session...");
      const zip = new AdmZip(buffer);

      // Extract into temp folder
      const sessionPath = path.join(__dirname, sessionName);
      if (fs.existsSync(sessionPath)) {
        fs.rmSync(sessionPath, { recursive: true, force: true });
      }
      zip.extractAllTo(sessionPath, true);

      // Step 3: Clone repo if not exists
      if (!fs.existsSync(BOT_FOLDER)) {
        console.log("📥 Cloning repository...");
        execSync(`git clone ${GITHUB_REPO} ${BOT_FOLDER}`, {
          stdio: "inherit",
        });
      } else {
        console.log("✅ Repo folder already exists, skipping clone.");
      }

      // Step 4: Install dependencies
      console.log("📦 Installing dependencies...");
      execSync(`cd ${BOT_FOLDER} && npm install`, { stdio: "inherit" });

      // Step 5: Move session into bot folder
      console.log("📂 Moving session into bot folder...");
      const targetSessionPath = path.join(BOT_FOLDER, "session");
      if (fs.existsSync(targetSessionPath)) {
        fs.rmSync(targetSessionPath, { recursive: true, force: true });
      }
      fs.renameSync(sessionPath, targetSessionPath);

      // Step 6: Start bot
      console.log("🚀 Starting ICEY MD Bot...");
      execSync(`cd ${BOT_FOLDER} && node index.js`, { stdio: "inherit" });
    } catch (err) {
      console.error("❌ Setup failed:", err.message);
    } finally {
      rl.close();
    }
  }
);
